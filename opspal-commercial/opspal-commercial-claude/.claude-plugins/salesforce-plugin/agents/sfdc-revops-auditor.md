```javascript
function assessAutomationComplexityWithHistory(flows, processBuilders, workflows, validationRules, context) {
  const automationPatterns = context.provenStrategies?.automationComplexity || {};

  // Calculate current complexity score
  const currentScore = {
    activeFlows: flows.filter(f => f.Status === 'Active').length,
    processBuilders: processBuilders.filter(pb => pb.IsActive).length,
    workflows: workflows.filter(w => w.IsActive).length,
    validationRules: validationRules.length,
    totalComplexity: 0
  };

  currentScore.totalComplexity =
    (currentScore.activeFlows * 3) +
    (currentScore.processBuilders * 2) +
    (currentScore.workflows * 1.5) +
    (currentScore.validationRules * 0.5);

  // Compare with historical patterns
  const historicalAvg = automationPatterns.averageComplexity || currentScore.totalComplexity;
  const historicalThresholds = automationPatterns.thresholds || {
    low: 50,
    medium: 150,
    high: 300
  };

  // Check for known automation anti-patterns
  const knownAntiPatterns = [];
  if (automationPatterns.commonIssues) {
    if (currentScore.activeFlows > (automationPatterns.commonIssues.maxFlowsBeforeIssues || 20)) {
      knownAntiPatterns.push({
        pattern: 'EXCESSIVE_FLOWS',
        severity: 'HIGH',
        historicalOccurrences: automationPatterns.commonIssues.excessiveFlowsCount || 0,
        recommendation: 'Consider flow consolidation - historical data shows orgs with >20 flows experience maintenance issues'
      });
    }

    if (currentScore.processBuilders > 0) {
      knownAntiPatterns.push({
        pattern: 'DEPRECATED_PROCESS_BUILDER',
        severity: 'MEDIUM',
        historicalOccurrences: automationPatterns.commonIssues.processBuilderCount || 0,
        recommendation: 'Migrate Process Builders to Flow - Salesforce has deprecated Process Builder'
      });
    }
  }

  return {
    currentComplexity: currentScore,
    complexityRating: currentScore.totalComplexity > historicalThresholds.high ? 'HIGH' :
                      currentScore.totalComplexity > historicalThresholds.medium ? 'MEDIUM' : 'LOW',
    vsHistorical: {
      difference: currentScore.totalComplexity - historicalAvg,
      percentile: calculatePercentile(currentScore.totalComplexity, automationPatterns.distribution || [])
    },
    antiPatterns: knownAntiPatterns,
    historicalInsights: automationPatterns.insights || []
  };
}

function calculatePercentile(value, distribution) {
  if (distribution.length === 0) return 50;
  const sorted = distribution.sort((a, b) => a - b);
  const below = sorted.filter(v => v < value).length;
  return Math.round((below / sorted.length) * 100);
}
```

```javascript
function analyzeFieldUtilizationWithBenchmarks(objectName, fields, recordCount, context) {
  const fieldBenchmarks = context.objectPatterns?.[objectName] || {};
  const utilization = [];

  for (const field of fields) {
    const fieldName = field.QualifiedApiName;

    // Query actual usage from org
    const usageQuery = `SELECT COUNT() FROM ${objectName} WHERE ${fieldName} != null`;
    const usageResult = await mcp_salesforce_data_query(usageQuery);
    const populatedCount = usageResult[0]?.expr0 || 0;
    const populationRate = recordCount > 0 ? (populatedCount / recordCount) * 100 : 0;

    // Compare with historical benchmark
    const benchmark = fieldBenchmarks.fieldUtilization?.[fieldName] || {};
    const historicalAvg = benchmark.avgPopulationRate || 50;
    const historicalMedian = benchmark.medianPopulationRate || 50;

    utilization.push({
      field: fieldName,
      populationRate: populationRate,
      populatedRecords: populatedCount,
      totalRecords: recordCount,
      vsHistorical: {
        averageDiff: populationRate - historicalAvg,
        medianDiff: populationRate - historicalMedian,
        isOutlier: Math.abs(populationRate - historicalMedian) > 30
      },
      recommendation: populationRate < 5 ?
        `Consider removing - only ${populationRate.toFixed(1)}% populated (historical avg: ${historicalAvg.toFixed(1)}%)` :
        populationRate < 20 ?
        `Low utilization - ${populationRate.toFixed(1)}% populated (historical avg: ${historicalAvg.toFixed(1)}%)` :
        `Healthy utilization - ${populationRate.toFixed(1)}% populated`
    });
  }

  // Sort by utilization rate
  utilization.sort((a, b) => a.populationRate - b.populationRate);

  return {
    objectName: objectName,
    totalFields: fields.length,
    utilization: utilization,
    lowUtilizationFields: utilization.filter(u => u.populationRate < 20).length,
    unusedFields: utilization.filter(u => u.populationRate === 0).length,
    outlierFields: utilization.filter(u => u.vsHistorical.isOutlier).length
  };
}
```

```javascript
function calculateRevOpsAssessmentConfidence(context, instanceType) {
  let confidence = 70; // Base confidence
  const factors = [];

  // Factor 1: Historical operation count
  if (context.totalOperations > 10) {
    confidence += 15;
    factors.push(`✅ Strong history: ${context.totalOperations} past assessments`);
  } else if (context.totalOperations > 5) {
    confidence += 8;
    factors.push(`⚠️  Moderate history: ${context.totalOperations} past assessments`);
  } else {
    factors.push(`❌ Limited history: ${context.totalOperations} past assessments`);
  }

  // Factor 2: Proven strategies available
  const provenStrategiesCount = Object.keys(context.provenStrategies || {}).length;
  if (provenStrategiesCount > 3) {
    confidence += 10;
    factors.push(`✅ ${provenStrategiesCount} proven strategies available`);
  }

  // Factor 3: Instance type (production vs sandbox)
  if (instanceType === 'PRODUCTION') {
    confidence += 5;
    factors.push('✅ Production instance - full data analysis possible');
  } else {
    confidence -= 10;
    factors.push('⚠️  Sandbox instance - metadata-only analysis');
  }

  // Factor 4: Object-specific patterns
  const objectPatternsCount = Object.keys(context.objectPatterns || {}).length;
  if (objectPatternsCount > 4) {
    confidence += 5;
    factors.push(`✅ Patterns for ${objectPatternsCount} objects`);
  }

  return {
    confidence: Math.min(100, Math.max(0, confidence)),
    level: confidence >= 85 ? 'HIGH' : confidence >= 70 ? 'MEDIUM' : 'LOW',
    factors: factors
  };
}
```

Include these expansion metrics for retention analysis:

| Metric | Formula | Benchmark (Good) |
|--------|---------|-----------------|

| **ACV Growth** | (ACV_current - ACV_previous) / ACV_previous | >10% YoY |
| **Retention/Expansion Mix** | Expansion Revenue / Total Retained Revenue | 30-50% |

## Troubleshooting

### Common Issues
1. **Insufficient Data Sample Size**
   - Symptoms: Statistical significance warnings, unreliable patterns
   - Solution: Extend timeframe or adjust sampling methodology
   - Prevention: Verify data volume before analysis initiation

2. **Integration Data Gaps**
   - Symptoms: Incomplete attribution tracking, missing source data
   - Solution: Implement additional tracking fields, review integration mappings
   - Prevention: Regular integration health monitoring

   - Symptoms: Inconsistent usage patterns, unclear adoption metrics
   - Solution: Define standardized adoption criteria, implement tracking mechanisms
   - Prevention: Establish baseline metrics during implementation
