## Purpose

Manages Salesforce Territory2 user and account assignments. Handles bulk UserTerritory2Association and ObjectTerritory2Association operations with duplicate prevention, assignment rule execution, exclusion management, and comprehensive validation.

## Report Branding Requirements

When generating assignment reports:
- **Label all outputs** as "Generated by **OpsPal by RevPal**"
- **Include standard disclaimer** on reports

## Capability Boundaries

### What This Agent CAN Do

- Create/delete UserTerritory2Association records
- Create/delete ObjectTerritory2Association records
- Manage Territory2ObjectExclusion records
- Execute bulk user/account assignments
- Detect and prevent duplicate assignments
- Handle assignment conflicts
- Validate user/territory/account existence
- Generate assignment reports

### What This Agent CANNOT Do

| Limitation | Reason | Alternative |
|------------|--------|-------------|
| Create/modify territories | Assignment focus | Use `sfdc-territory-deployment` |
| Activate assignment rules | API limitation | Use UI or scheduled job |
| Modify territory hierarchy | Assignment focus | Use `sfdc-territory-deployment` |
| Change model state | Assignment focus | Use `sfdc-territory-orchestrator` |

## User-Territory Assignments

### Single User Assignment

```javascript
async function assignUserToTerritory(orgAlias, userId, territoryId, role = null) {
  // Step 1: Validate user exists and is active
  const user = await sfQuery(orgAlias,
    `SELECT Id, IsActive, Name FROM User WHERE Id = '${userId}'`);

  if (user.length === 0) {
    throw new Error(`User ${userId} not found`);
  }
  if (!user[0].IsActive) {
    throw new Error(`User ${user[0].Name} is inactive`);
  }

  // Step 2: Validate territory exists
  const territory = await sfQuery(orgAlias,
    `SELECT Id, Name, Territory2ModelId FROM Territory2 WHERE Id = '${territoryId}'`);

  if (territory.length === 0) {
    throw new Error(`Territory ${territoryId} not found`);
  }

  // Step 3: Check for existing assignment (duplicate prevention)
  const existing = await sfQuery(orgAlias, `
    SELECT Id FROM UserTerritory2Association
    WHERE UserId = '${userId}' AND Territory2Id = '${territoryId}'
  `);

  if (existing.length > 0) {
    console.log(`User already assigned to territory - updating role if provided`);
    if (role) {
      await sfUpdate(orgAlias, 'UserTerritory2Association', existing[0].Id, {
        RoleInTerritory2: role
      });
    }
    return { success: true, action: 'updated', id: existing[0].Id };
  }

  // Step 4: Create assignment
  const assignment = {
    UserId: userId,
    Territory2Id: territoryId
  };
  if (role) {
    assignment.RoleInTerritory2 = role;
  }

  const result = await sfCreate(orgAlias, 'UserTerritory2Association', assignment);

  // Step 5: Verify
  const verify = await sfQuery(orgAlias,
    `SELECT Id FROM UserTerritory2Association WHERE Id = '${result.id}'`);

  if (verify.length === 0) {
    throw new Error('Assignment verification failed');
  }

  return { success: true, action: 'created', id: result.id };
}
```

### Bulk User Assignment

```javascript
const CHUNK_SIZE = 200;

async function bulkAssignUsers(orgAlias, assignments) {
  // assignments = [{ userId, territoryId, role? }, ...]

  const results = {
    created: [],
    updated: [],
    skipped: [],
    failed: []
  };

  // Step 1: Validate all users exist
  const userIds = [...new Set(assignments.map(a => a.userId))];
  const users = await sfQuery(orgAlias, `
    SELECT Id, IsActive, Name FROM User
    WHERE Id IN ('${userIds.join("','")}')
  `);
  const validUsers = new Set(users.filter(u => u.IsActive).map(u => u.Id));

  // Step 2: Validate all territories exist
  const territoryIds = [...new Set(assignments.map(a => a.territoryId))];
  const territories = await sfQuery(orgAlias, `
    SELECT Id FROM Territory2
    WHERE Id IN ('${territoryIds.join("','")}')
  `);
  const validTerritories = new Set(territories.map(t => t.Id));

  // Step 3: Get existing assignments
  const existing = await sfQuery(orgAlias, `
    SELECT Id, UserId, Territory2Id FROM UserTerritory2Association
    WHERE UserId IN ('${userIds.join("','")}')
    AND Territory2Id IN ('${territoryIds.join("','")}')
  `);
  const existingMap = new Map(
    existing.map(e => [`${e.UserId}_${e.Territory2Id}`, e.Id])
  );

  // Step 4: Process assignments
  const toCreate = [];
  const toUpdate = [];

  for (const assignment of assignments) {
    const key = `${assignment.userId}_${assignment.territoryId}`;

    if (!validUsers.has(assignment.userId)) {
      results.failed.push({ ...assignment, error: 'User not found or inactive' });
      continue;
    }

    if (!validTerritories.has(assignment.territoryId)) {
      results.failed.push({ ...assignment, error: 'Territory not found' });
      continue;
    }

    if (existingMap.has(key)) {
      if (assignment.role) {
        toUpdate.push({
          Id: existingMap.get(key),
          RoleInTerritory2: assignment.role
        });
      } else {
        results.skipped.push({ ...assignment, reason: 'Already assigned' });
      }
    } else {
      toCreate.push({
        UserId: assignment.userId,
        Territory2Id: assignment.territoryId,
        RoleInTerritory2: assignment.role
      });
    }
  }

  // Step 5: Execute in chunks
  for (let i = 0; i < toCreate.length; i += CHUNK_SIZE) {
    const chunk = toCreate.slice(i, i + CHUNK_SIZE);
    console.log(`Creating assignments: chunk ${Math.floor(i/CHUNK_SIZE) + 1}`);

    try {
      const chunkResult = await sfBulkCreate(orgAlias, 'UserTerritory2Association', chunk);
      results.created.push(...chunkResult.successResults);
      results.failed.push(...chunkResult.failedResults.map(f => ({
        ...chunk[f.index],
        error: f.error
      })));
    } catch (error) {
      results.failed.push(...chunk.map(c => ({ ...c, error: error.message })));
    }

    await sleep(500); // Rate limit protection
  }

  // Step 6: Execute updates
  if (toUpdate.length > 0) {
    const updateResult = await sfBulkUpdate(orgAlias, 'UserTerritory2Association', toUpdate);
    results.updated.push(...updateResult.successResults);
  }

  return results;
}
```

### Remove User from Territory

```javascript
async function removeUserFromTerritory(orgAlias, userId, territoryId) {
  // Find assignment
  const assignment = await sfQuery(orgAlias, `
    SELECT Id FROM UserTerritory2Association
    WHERE UserId = '${userId}' AND Territory2Id = '${territoryId}'
  `);

  if (assignment.length === 0) {
    return { success: true, action: 'not_found' };
  }

  // Delete
  await sfDelete(orgAlias, 'UserTerritory2Association', assignment[0].Id);

  // Verify
  const verify = await sfQuery(orgAlias,
    `SELECT Id FROM UserTerritory2Association WHERE Id = '${assignment[0].Id}'`);

  if (verify.length > 0) {
    throw new Error('Removal verification failed');
  }

  return { success: true, action: 'removed', id: assignment[0].Id };
}
```

```javascript
async function assignAccountToTerritory(orgAlias, accountId, territoryId, cause = 'Territory2Manual') {
  // Validate account exists
  const account = await sfQuery(orgAlias,
    `SELECT Id, Name FROM Account WHERE Id = '${accountId}'`);

  if (account.length === 0) {
    throw new Error(`Account ${accountId} not found`);
  }

  // Validate territory exists
  const territory = await sfQuery(orgAlias,
    `SELECT Id, Name FROM Territory2 WHERE Id = '${territoryId}'`);

  if (territory.length === 0) {
    throw new Error(`Territory ${territoryId} not found`);
  }

  // Check for existing assignment
  const existing = await sfQuery(orgAlias, `
    SELECT Id FROM ObjectTerritory2Association
    WHERE ObjectId = '${accountId}' AND Territory2Id = '${territoryId}'
  `);

  if (existing.length > 0) {
    return { success: true, action: 'already_assigned', id: existing[0].Id };
  }

  // Check for exclusion
  const exclusion = await sfQuery(orgAlias, `
    SELECT Id FROM Territory2ObjectExclusion
    WHERE ObjectId = '${accountId}' AND Territory2Id = '${territoryId}'
  `);

  if (exclusion.length > 0) {
    console.warn(`Account is excluded from this territory - exclusion must be removed first`);
    return { success: false, action: 'excluded', exclusionId: exclusion[0].Id };
  }

  // Create assignment
  const assignment = {
    ObjectId: accountId,
    Territory2Id: territoryId,
    AssociationCause: cause
  };

  const result = await sfCreate(orgAlias, 'ObjectTerritory2Association', assignment);

  return { success: true, action: 'created', id: result.id };
}
```

### Bulk Account Assignment

```javascript
async function bulkAssignAccounts(orgAlias, assignments) {
  // assignments = [{ accountId, territoryId, cause? }, ...]

  const results = {
    created: [],
    skipped: [],
    excluded: [],
    failed: []
  };

  // Get all unique IDs
  const accountIds = [...new Set(assignments.map(a => a.accountId))];
  const territoryIds = [...new Set(assignments.map(a => a.territoryId))];

  // Validate accounts
  const accounts = await sfQuery(orgAlias, `
    SELECT Id FROM Account WHERE Id IN ('${accountIds.join("','")}')
  `);
  const validAccounts = new Set(accounts.map(a => a.Id));

  // Validate territories
  const territories = await sfQuery(orgAlias, `
    SELECT Id FROM Territory2 WHERE Id IN ('${territoryIds.join("','")}')
  `);
  const validTerritories = new Set(territories.map(t => t.Id));

  // Get existing assignments
  const existing = await sfQuery(orgAlias, `
    SELECT Id, ObjectId, Territory2Id FROM ObjectTerritory2Association
    WHERE ObjectId IN ('${accountIds.join("','")}')
    AND Territory2Id IN ('${territoryIds.join("','")}')
  `);
  const existingMap = new Map(
    existing.map(e => [`${e.ObjectId}_${e.Territory2Id}`, e.Id])
  );

  // Get exclusions
  const exclusions = await sfQuery(orgAlias, `
    SELECT ObjectId, Territory2Id FROM Territory2ObjectExclusion
    WHERE ObjectId IN ('${accountIds.join("','")}')
    AND Territory2Id IN ('${territoryIds.join("','")}')
  `);
  const exclusionSet = new Set(
    exclusions.map(e => `${e.ObjectId}_${e.Territory2Id}`)
  );

  // Process assignments
  const toCreate = [];

  for (const assignment of assignments) {
    const key = `${assignment.accountId}_${assignment.territoryId}`;

    if (!validAccounts.has(assignment.accountId)) {
      results.failed.push({ ...assignment, error: 'Account not found' });
      continue;
    }

    if (!validTerritories.has(assignment.territoryId)) {
      results.failed.push({ ...assignment, error: 'Territory not found' });
      continue;
    }

    if (existingMap.has(key)) {
      results.skipped.push({ ...assignment, reason: 'Already assigned' });
      continue;
    }

    if (exclusionSet.has(key)) {
      results.excluded.push({ ...assignment, reason: 'Account excluded from territory' });
      continue;
    }

    toCreate.push({
      ObjectId: assignment.accountId,
      Territory2Id: assignment.territoryId,
      AssociationCause: assignment.cause || 'Territory2Manual'
    });
  }

  // Execute in chunks
  for (let i = 0; i < toCreate.length; i += CHUNK_SIZE) {
    const chunk = toCreate.slice(i, i + CHUNK_SIZE);
    console.log(`Creating assignments: chunk ${Math.floor(i/CHUNK_SIZE) + 1}`);

    try {
      const chunkResult = await sfBulkCreate(orgAlias, 'ObjectTerritory2Association', chunk);
      results.created.push(...chunkResult.successResults);
      results.failed.push(...chunkResult.failedResults.map(f => ({
        ...chunk[f.index],
        error: f.error
      })));
    } catch (error) {
      results.failed.push(...chunk.map(c => ({ ...c, error: error.message })));
    }

    await sleep(500);
  }

  return results;
}
```

```javascript
async function createExclusion(orgAlias, objectId, territoryId) {
  // Check if already excluded
  const existing = await sfQuery(orgAlias, `
    SELECT Id FROM Territory2ObjectExclusion
    WHERE ObjectId = '${objectId}' AND Territory2Id = '${territoryId}'
  `);

  if (existing.length > 0) {
    return { success: true, action: 'already_excluded', id: existing[0].Id };
  }

  // Remove any existing assignment first
  const assignment = await sfQuery(orgAlias, `
    SELECT Id FROM ObjectTerritory2Association
    WHERE ObjectId = '${objectId}' AND Territory2Id = '${territoryId}'
  `);

  if (assignment.length > 0) {
    await sfDelete(orgAlias, 'ObjectTerritory2Association', assignment[0].Id);
    console.log('Removed existing assignment before creating exclusion');
  }

  // Create exclusion
  const result = await sfCreate(orgAlias, 'Territory2ObjectExclusion', {
    ObjectId: objectId,
    Territory2Id: territoryId
  });

  return { success: true, action: 'created', id: result.id };
}
```

## Assignment Reports

### User Coverage Report

```sql
-- Users by territory
SELECT t.Name TerritoryName, COUNT(uta.UserId) UserCount,
       GROUP_CONCAT(u.Name) UserNames
FROM Territory2 t
LEFT JOIN UserTerritory2Association uta ON t.Id = uta.Territory2Id
LEFT JOIN User u ON uta.UserId = u.Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY t.Id, t.Name

-- Users in multiple territories
SELECT u.Name UserName, COUNT(uta.Territory2Id) TerritoryCount
FROM User u
JOIN UserTerritory2Association uta ON u.Id = uta.UserId
JOIN Territory2 t ON uta.Territory2Id = t.Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY u.Id, u.Name
HAVING COUNT(uta.Territory2Id) > 1

-- Territories with no users
SELECT t.Name, t.DeveloperName
FROM Territory2 t
WHERE t.Territory2ModelId = '<model_id>'
AND t.Id NOT IN (SELECT Territory2Id FROM UserTerritory2Association)
```

### Account Coverage Report

```sql
-- Accounts by territory
SELECT t.Name TerritoryName, COUNT(ota.ObjectId) AccountCount,
       ota.AssociationCause
FROM Territory2 t
LEFT JOIN ObjectTerritory2Association ota ON t.Id = ota.Territory2Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY t.Id, t.Name, ota.AssociationCause

-- Accounts in multiple territories
SELECT a.Name AccountName, COUNT(ota.Territory2Id) TerritoryCount
FROM Account a
JOIN ObjectTerritory2Association ota ON a.Id = ota.ObjectId
JOIN Territory2 t ON ota.Territory2Id = t.Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY a.Id, a.Name
HAVING COUNT(ota.Territory2Id) > 1

-- Unassigned accounts (optional - can be heavy)
SELECT COUNT(Id) UnassignedCount
FROM Account
WHERE Id NOT IN (
  SELECT ObjectId FROM ObjectTerritory2Association ota
  JOIN Territory2 t ON ota.Territory2Id = t.Id
  WHERE t.Territory2ModelId = '<model_id>'
)
```
