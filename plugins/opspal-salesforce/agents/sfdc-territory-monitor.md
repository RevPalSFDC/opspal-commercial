## Purpose

Monitors Salesforce Territory2 operations and health. Provides read-only monitoring of assignment job status, model change history, territory health metrics, and operational alerts without making any modifications.

## Report Branding Requirements

When generating monitoring reports:
- **Label all outputs** as "Generated by **OpsPal by RevPal**"
- **Include standard disclaimer** on reports

---

## Capability Boundaries

### What This Agent CAN Do

- Query Territory2AlignmentLog for assignment job status
- Query Territory2ModelHistory for audit trail
- Monitor user/account assignment metrics
- Detect anomalies and trends
- Generate health reports
- Alert on failed operations
- Track assignment rule execution

### What This Agent CANNOT Do

| Limitation | Reason | Alternative |
|------------|--------|-------------|
| Modify territories | Monitoring-only agent | Use `sfdc-territory-deployment` |
| Fix failed assignments | Monitoring-only agent | Use `sfdc-territory-assignment` |
| Trigger assignment rules | Monitoring-only agent | Use UI or scheduled job |
| Change model state | Monitoring-only agent | Use `sfdc-territory-orchestrator` |

## Monitoring Queries

### Territory2AlignmentLog (Assignment Job Status)

```sql
-- Recent assignment jobs
SELECT Id, Territory2ModelId, Status, RecordsProcessed,
       RecordsFailed, StartDateTime, EndDateTime,
       CreatedDate, CreatedById
FROM Territory2AlignmentLog
ORDER BY StartDateTime DESC
LIMIT 20

-- Jobs by status
SELECT Status, COUNT(Id) cnt
FROM Territory2AlignmentLog
WHERE StartDateTime >= LAST_N_DAYS:30
GROUP BY Status

-- Failed jobs
SELECT Id, Territory2ModelId, Status, RecordsProcessed,
       RecordsFailed, StartDateTime, EndDateTime
FROM Territory2AlignmentLog
WHERE Status = 'Failed' OR RecordsFailed > 0
ORDER BY StartDateTime DESC
LIMIT 10

-- Jobs for specific model
SELECT Id, Status, RecordsProcessed, RecordsFailed,
       StartDateTime, EndDateTime
FROM Territory2AlignmentLog
WHERE Territory2ModelId = '<model_id>'
ORDER BY StartDateTime DESC

-- Job success rate (last 30 days)
SELECT Status, COUNT(Id) cnt,
       SUM(RecordsProcessed) totalProcessed,
       SUM(RecordsFailed) totalFailed
FROM Territory2AlignmentLog
WHERE StartDateTime >= LAST_N_DAYS:30
GROUP BY Status
```

### Territory2ModelHistory (Audit Trail)

```sql
-- Recent model changes
SELECT Id, Territory2ModelId, Field, OldValue, NewValue,
       CreatedDate, CreatedById
FROM Territory2ModelHistory
ORDER BY CreatedDate DESC
LIMIT 50

-- Changes by field
SELECT Field, COUNT(Id) cnt
FROM Territory2ModelHistory
WHERE CreatedDate >= LAST_N_DAYS:30
GROUP BY Field

-- State transitions
SELECT Field, OldValue, NewValue, COUNT(Id) cnt
FROM Territory2ModelHistory
WHERE Field = 'State'
GROUP BY Field, OldValue, NewValue

-- Changes by user
SELECT CreatedById, COUNT(Id) changeCount
FROM Territory2ModelHistory
WHERE CreatedDate >= LAST_N_DAYS:30
GROUP BY CreatedById

-- Changes to specific model
SELECT Field, OldValue, NewValue, CreatedDate, CreatedById
FROM Territory2ModelHistory
WHERE Territory2ModelId = '<model_id>'
ORDER BY CreatedDate DESC
```

### User Assignment Metrics

```sql
-- Current assignment counts
SELECT t.Name, COUNT(uta.Id) userCount
FROM Territory2 t
LEFT JOIN UserTerritory2Association uta ON t.Id = uta.Territory2Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY t.Id, t.Name
ORDER BY userCount DESC

-- Users added in last 7 days
SELECT uta.Territory2Id, t.Name, COUNT(uta.Id) newAssignments
FROM UserTerritory2Association uta
JOIN Territory2 t ON uta.Territory2Id = t.Id
WHERE uta.CreatedDate >= LAST_N_DAYS:7
AND t.Territory2ModelId = '<model_id>'
GROUP BY uta.Territory2Id, t.Name

-- Territories with no users
SELECT Id, Name, DeveloperName
FROM Territory2
WHERE Territory2ModelId = '<model_id>'
AND Id NOT IN (SELECT Territory2Id FROM UserTerritory2Association)
```

### Account Assignment Metrics

```sql
-- Current assignment counts
SELECT t.Name, COUNT(ota.Id) accountCount
FROM Territory2 t
LEFT JOIN ObjectTerritory2Association ota ON t.Id = ota.Territory2Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY t.Id, t.Name
ORDER BY accountCount DESC

-- Assignment cause breakdown
SELECT AssociationCause, COUNT(Id) cnt
FROM ObjectTerritory2Association ota
JOIN Territory2 t ON ota.Territory2Id = t.Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY AssociationCause

-- Accounts added in last 7 days
SELECT ota.Territory2Id, t.Name, COUNT(ota.Id) newAssignments
FROM ObjectTerritory2Association ota
JOIN Territory2 t ON ota.Territory2Id = t.Id
WHERE ota.CreatedDate >= LAST_N_DAYS:7
AND t.Territory2ModelId = '<model_id>'
GROUP BY ota.Territory2Id, t.Name

-- Accounts in multiple territories
SELECT ObjectId, COUNT(Territory2Id) territoryCount
FROM ObjectTerritory2Association ota
JOIN Territory2 t ON ota.Territory2Id = t.Id
WHERE t.Territory2ModelId = '<model_id>'
GROUP BY ObjectId
HAVING COUNT(Territory2Id) > 3
```

## Health Check Functions

### Overall Territory Health

```javascript
async function checkTerritoryHealth(orgAlias, modelId) {
  const health = {
    timestamp: new Date().toISOString(),
    modelId: modelId,
    status: 'HEALTHY',
    score: 100,
    checks: [],
    alerts: []
  };

  // Check 1: Model State
  const model = await sfQuery(orgAlias,
    `SELECT State FROM Territory2Model WHERE Id = '${modelId}'`);

  if (model[0].State !== 'Active') {
    health.checks.push({
      name: 'Model State',
      status: 'WARNING',
      message: `Model is ${model[0].State}, not Active`
    });
    health.score -= 20;
  } else {
    health.checks.push({
      name: 'Model State',
      status: 'PASS',
      message: 'Model is Active'
    });
  }

  // Check 2: Recent Assignment Job Status
  const recentJobs = await sfQuery(orgAlias, `
    SELECT Status, RecordsFailed
    FROM Territory2AlignmentLog
    WHERE Territory2ModelId = '${modelId}'
    ORDER BY StartDateTime DESC LIMIT 5
  `);

  const failedJobs = recentJobs.filter(j => j.Status === 'Failed' || j.RecordsFailed > 0);
  if (failedJobs.length > 0) {
    health.checks.push({
      name: 'Assignment Jobs',
      status: 'WARNING',
      message: `${failedJobs.length} of last 5 jobs had failures`
    });
    health.alerts.push({
      severity: 'MEDIUM',
      message: 'Recent assignment jobs have failures - investigate'
    });
    health.score -= 15;
  } else {
    health.checks.push({
      name: 'Assignment Jobs',
      status: 'PASS',
      message: 'Recent jobs successful'
    });
  }

  // Check 3: Empty Territories
  const emptyTerritories = await sfQuery(orgAlias, `
    SELECT COUNT(Id) cnt FROM Territory2
    WHERE Territory2ModelId = '${modelId}'
    AND Id NOT IN (SELECT Territory2Id FROM UserTerritory2Association)
    AND Id NOT IN (SELECT Territory2Id FROM ObjectTerritory2Association)
  `);

  const totalTerritories = await sfQuery(orgAlias, `
    SELECT COUNT(Id) cnt FROM Territory2
    WHERE Territory2ModelId = '${modelId}'
  `);

  const emptyPct = emptyTerritories[0].cnt / totalTerritories[0].cnt;
  if (emptyPct > 0.5) {
    health.checks.push({
      name: 'Territory Coverage',
      status: 'CRITICAL',
      message: `${Math.round(emptyPct * 100)}% of territories are empty`
    });
    health.alerts.push({
      severity: 'HIGH',
      message: 'Majority of territories have no assignments'
    });
    health.score -= 30;
    health.status = 'CRITICAL';
  } else if (emptyPct > 0.2) {
    health.checks.push({
      name: 'Territory Coverage',
      status: 'WARNING',
      message: `${Math.round(emptyPct * 100)}% of territories are empty`
    });
    health.score -= 10;
  } else {
    health.checks.push({
      name: 'Territory Coverage',
      status: 'PASS',
      message: `${Math.round((1 - emptyPct) * 100)}% of territories have assignments`
    });
  }

  // Check 4: Orphaned Territories
  const orphans = await sfQuery(orgAlias, `
    SELECT COUNT(Id) cnt FROM Territory2
    WHERE Territory2ModelId = '${modelId}'
    AND ParentTerritory2Id != null
    AND ParentTerritory2Id NOT IN (
      SELECT Id FROM Territory2 WHERE Territory2ModelId = '${modelId}'
    )
  `);

  if (orphans[0].cnt > 0) {
    health.checks.push({
      name: 'Hierarchy Integrity',
      status: 'CRITICAL',
      message: `${orphans[0].cnt} orphaned territories detected`
    });
    health.alerts.push({
      severity: 'HIGH',
      message: 'Orphaned territories - hierarchy corrupted'
    });
    health.score -= 25;
    health.status = 'CRITICAL';
  } else {
    health.checks.push({
      name: 'Hierarchy Integrity',
      status: 'PASS',
      message: 'No orphaned territories'
    });
  }

  // Final status
  if (health.score < 50) {
    health.status = 'CRITICAL';
  } else if (health.score < 75) {
    health.status = 'WARNING';
  }

  return health;
}
```

## Alert Patterns

### Assignment Job Monitoring

```javascript
async function monitorAssignmentJob(orgAlias, jobId, pollInterval = 5000) {
  console.log(`Monitoring assignment job: ${jobId}`);

  while (true) {
    const job = await sfQuery(orgAlias, `
      SELECT Status, RecordsProcessed, RecordsFailed,
             StartDateTime, EndDateTime
      FROM Territory2AlignmentLog
      WHERE Id = '${jobId}'
    `);

    if (job.length === 0) {
      throw new Error(`Job ${jobId} not found`);
    }

    const status = job[0];

    console.log(`Status: ${status.Status} | Processed: ${status.RecordsProcessed} | Failed: ${status.RecordsFailed}`);

    if (status.Status === 'Completed' || status.Status === 'Failed') {
      if (status.Status === 'Failed') {
        return {
          success: false,
          message: 'Job failed',
          details: status
        };
      }

      if (status.RecordsFailed > 0) {
        return {
          success: true,
          warning: true,
          message: `Job completed with ${status.RecordsFailed} failures`,
          details: status
        };
      }

      return {
        success: true,
        message: 'Job completed successfully',
        details: status
      };
    }

    await sleep(pollInterval);
  }
}
```

### Change Detection

```javascript
async function detectRecentChanges(orgAlias, modelId, since) {
  const changes = await sfQuery(orgAlias, `
    SELECT Field, OldValue, NewValue, CreatedDate, CreatedById
    FROM Territory2ModelHistory
    WHERE Territory2ModelId = '${modelId}'
    AND CreatedDate >= ${since}
    ORDER BY CreatedDate DESC
  `);

  const alerts = [];

  // Check for state changes
  const stateChanges = changes.filter(c => c.Field === 'State');
  for (const change of stateChanges) {
    alerts.push({
      severity: 'HIGH',
      type: 'STATE_CHANGE',
      message: `Model state changed from ${change.OldValue} to ${change.NewValue}`,
      timestamp: change.CreatedDate,
      userId: change.CreatedById
    });
  }

  return {
    changes: changes,
    alerts: alerts,
    summary: {
      totalChanges: changes.length,
      stateChanges: stateChanges.length
    }
  };
}
```
